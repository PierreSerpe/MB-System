37 /*
38  * Notes on the MBSYS_HYSWEEP HYPACK and HYSWEEP data:
39  * 1. The raw HYPACK and HYSWEEP data formats are associated with HYPACK software
40  * from Hypack Inc. of Middletown, CT, USA, formally known as Coastal
41  * Oceanographics. The raw format produced by the Hypack dataloggers
42  * is constructed of pure ASCII text.
43  *
44  * The following information is take more or less verbatim from the
45  * 2010 HSX format specification.
46  *
47  * Coastal Oceanographics, Inc.
48  * Technical Note:
49  * HSX Format - Hysweep Text (ASCII) Logging
50  * Hysweep survey has a Text logging option (HSX format), allowing raw data
51  * to be stored in a format that can be inspected and modified by most
52  * editing program (Windows Wordpad for example). Easy inspection of files
53  * is the advantage of text logging - the disadvantage is larger files and
54  * slower load time. If file size and load time are important to you, it is
55  * best to choose the Hysweep binary format (HS2).
56  *
57  * HSX files are generally compatible with Hypack Survey raw format, allowing
58  * Hypack programs (Hypack Max, Hyplot, etc.) to work with HSX files. The
59  * differences involve logging and processing of multibeam data, which is
60  * by the Hysweep extensions to Hypack.
61  *
62  * Each file contains two sections; a header, which is written when data
63  * logging starts, and a data section, which is written as data is collected.
64  * Most records starts with a three character tag.
65  *
66  * Header Tags:
67  * DEV Hypack Device Information
68  * DV2 Hysweep Device Information
69  * EOH End of Header
70  * EOL End of Planned Line
71  * FTP File Type (Hypack File Identifier)
72  * HSP Hysweep Survey Parameters
73  * HSX HSX File Identifier
74  * HVF Hysweep View Filters
75  * INF General Information
76  * LBP Planned Line Begin Point
77  * LIN Planned Line Data follows
78  * LNN Planned Line Name
79  * MBI Multibeam / Multiple Transducer Device information
80  * 0F2 Hysweep Device Offsets
81  * PRI Primary Navigation Device
82  * PTS Planned Line Waypoint
83  * SSI Sidescan Device Information
84  * SVC Sound Velocity Correction
85  * TND Survey Time and Date
86  * PRJ Projected coordinate system - MB-System extension
87  *
88  * Data Tags:
89  * DFT Dynamic Draft (Squat) Correction
90  * FIX Fix (Event) Mark
91  * HCP Heave Compensation
92  * EC1 Echo Sounding (single frequency)
93  * GPS GPS Measurements
94  * GYR Gyro Data (Heading)
95  * POS Position
96  * PSA Pitch Stabilization Angle.
97  * RMB Raw Multibeam data
98  * RSS Raw Sidescan data
99  * SNR Sonar Runtime Settings
100  * TID Tide Correction
101  * COM Comment - MB-System extension
102  * PRJ Projected coordinate system definition for eastings and
103  * northings in file - MB-System extension
104  * MSS Sidescan laid out on seafloor - MB-System extension
105  *
106  */

259  /* DEV: first line of device declaration
260  DEV dn dc "name"
261  DEV 0 544 "IXSEA OCTANS Serial" */
262  int DEV_device_number;
263  int DEV_device_capability; /* Hypack device capabilities (bit code)
264  1 - device provides range/range position
265  2 - device provides range/azimuth position
266  4 - device provides lat/long (e.g. GPS)
267  8 - device provides grid positions XY
268  16 - device provides echo soundings
269  32 - device provides heading
270  64 - device provides ship speed
271  128 - Hypack clock is synched to device clock
272  256 - device provides tide
273  512 - device provides heave, pitch, and roll
274  1024 - device is a ROV
275  2048 - device is a left/right indicator
276  4096 - device accepts annotations strings
277  8192 - device accepts output from Hypack
278  16384 - xxx
279  32768 - device has extended capability
280  NOTE: an MB-System extension to allow */
281  mb_path DEV_device_name; /* e.g. "GPS" */
282 
283  /* DV2: second line of device declaration
284  DV2 dn dc tf en
285  DV2 0 220 0 1 */
286  int DV2_device_capability; /* Hysweep device capabilities (bit coded hexadecimal)
287  0001 - multibeam sonar
288  0002 - multiple transducer sonar
289  0004 - GPS (boat position)
290  0008 - sidescan sonar
291  0010 - single beam echosounder
292  0020 - gyro (boat heading)
293  0040 - tide
