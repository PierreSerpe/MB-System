25  * mbf_sburicen.h defines the data structure used by MBIO functions
26  * to store multibeam data read from the MBF_SBURICEN format (MBIO id 14).
27  *
28  * Author: D. W. Caress
29  * Date: January 20, 1993
30  *
31  */
32 /*
33  * Notes on the MBF_SBURICEN data format:
34  * 1. This data format is used to store 16 beam Sea Beam bathymetry
35  * data. This format was created and used by the Ocean Mapping
36  * Development Center at the Graduate School of Oceanography of
37  * the University of Rhode Islande; most data files in this format
38  * consist of Sea Beam data collected on the R/V Robert Conrad or
39  * the R/V Atlantis II.
40  * 2. The data consist of 102 byte records consisting entirely of
41  * 2-byte integers.
42  * 3. The 16 depth values are stored centered in 19 value arrays. The
43  * center beam is in word 10 of the depth and distance arrays.
44  * 4. Comments can be embedded in the data as 100 byte ascii strings,
45  * where the first two characters must always be "cc" so that the
46  * first depth value is 25443.
47  * 5. Information on this format was obtained from the Geological
48  * Data Center at the Scripps Institution of Oceanography
49  *
50  * The kind value in the mbf_sburicen_struct indicates whether the
51  * mbf_sburicen_data_struct structure holds data (kind = 1) or an
52  * ascii comment record (kind = 2).
53  *
54  * The mbf_sburicen_data_struct structure is a direct representation
55  * of the binary data structure used in the MBF_SBURICEN format.
56  */
57

61 struct mbf_sburicen_data_struct {
62  short deph[19]; /* 16 depths from Sea Beam in meters
63  assuming 1500 m/s water velocity */
64  short dist[19]; /* 16 cross track distances in meters from port
65  (negative) to starboard (positive) */
66  short axis; /* navigation error ellipse major axis angle */
67  short major; /* navigation error ellipse major axis */
68  short minor; /* navigation error ellipse minor axis */
69  unsigned short sbhdg; /* Sea Beam gyro heading
70  0 = 0 degrees
71  1 = 0.0055 degrees
72  16384 = 90 degrees
73  65535 = 359.9945 degrees
74  0 = 360 degrees */
75  short lat2b; /* fraction of minute times 10000 */
76  short lat2u; /* number of minutes north of 90S */
77  short lon2b; /* fraction of minute times 10000 */
78  short lon2u; /* minutes east of prime meridian */
79  short sec; /* seconds from beginning of minute (0-59) */
80  short min; /* minutes from beginning of day (0-1439) */
81  short day; /* julian day (1-366) */
82  short year; /* year (4 digits) */
83  unsigned short sbtim; /* Sea Beam computer clock time in 10ths of
84  seconds from start of hour (0-3600) */
85 };
