24 /*
25  * mbsys_simrad.h defines the MBIO data structures for handling data from
26  * old (pre-1997) Simrad multibeam sonars (e.g. EM950, EM1000, EM12S,
27  * EM12D). The data formats associated with Simrad multibeams
28  * (both old and new) include:
29  * MBSYS_SIMRAD formats (code in mbsys_simrad.c and mbsys_simrad.h):
30  * MBF_EMOLDRAW : MBIO ID 51 - Vendor EM1000, EM12S, EM12D, EM121
31  * : MBIO ID 52 - aliased to 51
32  * MBF_EM12IFRM : MBIO ID 53 - IFREMER EM12S and EM12D
33  * MBF_EM12DARW : MBIO ID 54 - NERC EM12S
34  * : MBIO ID 55 - aliased to 51
35  * MBSYS_SIMRAD2 formats (code in mbsys_simrad2.c and mbsys_simrad2.h):
36  * MBF_EM300RAW : MBIO ID 56 - Vendor EM3000, EM300, EM120
37  * MBF_EM300MBA : MBIO ID 57 - MBARI EM3000, EM300, EM120
38  *
39  *
40  * Author: D. W. Caress (L-DEO)
41  * Date: August 5, 1994
42  *
43  */
44 /*
45  * Notes on the MBSYS_SIMRAD data structure:
46  * 1. Simrad multibeam systems output datagrams which are
47  * a combination of ascii and binary.
48  * 2. Simrad multibeam sonars output both bathymetry
49  * and amplitude information for beams and sidescan information
50  * with a higher resolution than the bathymetry and amplitude.
51  * 3. There are five systems of interest:
52  * EM-1000: Shallow water system with up to 60 beams of
53  * bathymetry and up to 523 sidescan samples per
54  * bathymetry beam.
55  * EM-12S: Single array deep water system with up to 81
56  * beams of bathymetry and up to 523 sidescan
57  * samples per bathymetry beam.
58  * EM-12D: Double array deep water system with up to
59  * 81 beams of bathymetry (port and starboard
60  * calculated and recorded separately) and up
61  * to 523 sidescan samples per bathymetry beam.
62  * EM-121: Single array deep water system with up to 121
63  * beams of bathymetry and up to 523 sidescan
64  * samples per bathymetry beam.
65  * 4. Each telegram is preceded by a two byte start code and
66  * followed by a three byte end code consisting of 0x03
67  * followed by two bytes representing the checksum for
68  * the data bytes. MB-System does not calculate checksums
69  * and puts 0's in the checksum bytes.
70  * 5. The relevant telegram start codes, types, and sizes are:
71  * 0x0285: Start 421 data bytes
72  * 0x0286: Stop 421 data bytes
73  * 0x0287: Parameter 421 data bytes
74  * 0x0293: Position 90 data bytes
75  * 0x029A: Sound velocity profile 416 data bytes
76  * 0x0294: EM-12D starboard bathymetry 923 data bytes
77  * 0x0295: EM-12D port bathymetry 923 data bytes
78  * 0x0296: EM-12S bathymetry 923 data bytes
79  * 0x0288: EM-121 bathymetry 1375 data bytes
80  * 0x0297: EM-1000 bathymetry 692 data bytes
81  * 0x02C8: EM-12D port sidescan 551 data bytes
82  * 0x02C9: EM-12D starboard sidescan 551 data bytes
83  * 0x02CA: EM-12S or EM-1000 sidescan 551 data bytes
84  * 0x02CB: EM-12D port sidescan + phase 1465 data bytes
85  * 0x02CC: EM-12D starboard sidescan + phase 1465 data bytes
86  * 0x02CD: EM-12S or EM-1000 sidescan + phase 1465 data bytes
87  * 6. The EM-12D system records separate starboard and port datagrams
88  * for each ping.
89  * 7. Multiple sidescan datagrams are recorded for each ping because
90  * there is too much information to fit in a single datagram.
91  * 8. Simrad systems record navigation fixes using the position
92  * datagram; no navigation is included in the per ping data. Thus,
93  * it is necessary to extrapolate the navigation for each ping
94  * at read time from the last navigation fix. The frequency of
95  * GPS fixes generally assures that this is not a problem, but
96  * we offer no guarantees that this will always be the case.
97  *
98  */
