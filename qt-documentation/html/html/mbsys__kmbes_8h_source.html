35 /*
36  * Notes on the mbsys_kmbes data structure and associated format:
37  * 1. This MBIO format supports the generic interface message output
38  * by the Kongsberg EM datagram format.
39  * 2. Reference: Kongsberg EM datagram format Reg. no. 410224 rev F, November 2018
40  * Kongsberg EM datagram format Reg. nr. 410224 rev J, 15-09-2023
41  * 3. The Kongsberg EM datagram stream consists of several different data
42  * records that can vary among models and installations
43  * 4. The Kongsberg EM datagram format supports the following multibeam models
44  * September 2016:
45  * EM 710, 40-100 kHz
46  * EM 712, 40-100 kHz
47  * EM 2040, 200/300/400 kHz
48  * EM 2040C, 180-400 kHz
49  * September 2023:
50  *
51  * 5. Each datagram begins with a 20 byte sequence:
52  * unsigned int numBytesDgm; // Datagram length in bytes.
53  * mb_u_char dgmType[4]; // Multi beam datagram type definition, e.g. #AAA
54  * mb_u_char dgmVersion; // Datagram version.
55  * mb_u_char systemID; // System ID.
56  * unsigned short echoSounderID; // Echo sounder identity.
57  * unsigned int time_sec; // Time in second.
58  * unsigned int time_nanosec; // Nano seconds remainder.
59  * 5. Each datagram ends with:
60  * unsigned int numBytesDgm; // Datagram length in bytes.
61  * 6. All data are in little-endian form.
62  * 7. The data record names include:
63  * // I - datagrams (Installation and Runtime)
64  * IIP, // Installation parameters and sensor setup.
65  * IOP, // Runtime parameters as chosen by operator.
66  * IBE, // Built in test (BIST) error report
67  * IBR, // Built in test (BIST) reply
68  * IBS, // Built in test (BIST) short reply
69  *
70  * // S-datagrams (Sensors)
71  * SPO, // Sensor (S) data for position (PO)
72  * SPE, // Sensor (S) data for postion error (PE) statistics
73  * SPD, // Sensor (S) data for position datum reference (PD)
74  * SKM, // Sensor (S) KM binary sensor format
75  * SVP, // Sensor (S) data from sound velocity (V) profile (P) or CTD
76  * SVT, // Sensor (S) data for sound velocity (V) at transducer (T)
77  * SCL, // Sensor (S) data from clock (CL)
78  * SDE, // Sensor (S) data from depth (DE) sensor
79  * SHI, // Sensor (S) data for height (HI)
80  * SHA, // Sensor (S) data for heading (HA)
81  *
82  * // M-datagrams (Multibeam)
83  * MRZ, // Multibeam (M) raw range (R) and depth (Z) datagram
84  * MWC, // Multibeam (M) water (W) column (C) datagram
85  *
86  * // C-datagrams (Compatibility)
87  * CHE, // Compatibility (C) data for heave (HE) - store raw sensor data without modification
88  * CPO, // Compatibility (C) data for position (PO) - store raw sensor data without modification
89  *
90  * // F-datagrams (Files)
91  * FCF, // Backscatter calibration (C) file (F) datagram
92  *
93  * // X-datagrams (eXtra - defined only for MB-System)
94  * XMB, // The presence of this datagram indicates this file/stream has been
95  * // written by MB-System.
96  * // - this means that pings include a sidescan datagram XMS after the MRZ datagrams
97  * // - this means that MB-System beamflags are embedded in the MRZ datagram soundings
98  * XMC, // Comment datagram (MB-System only)
99  * XMS, // MB-System multibeam pseudosidescan derived from multibeam backscatter (MB-System only)
100  */
