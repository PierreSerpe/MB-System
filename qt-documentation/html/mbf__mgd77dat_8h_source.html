25  * mbf_mgd77dat.h defines the data structures used by MBIO functions
26  * to store multibeam data read from the MBF_MGD77DAT format (MBIO id 21).
27  *
28  * Author: D. W. Caress
29  * Date: May 19, 1999
30  *
31  *
32  */
33 /*
34  * Notes on the MBF_MGD77DAT data format:
35  * 1. The MGD77 format is is an exchange format for marine
36  * geophysical data (bathymetry, magnetics, and gravity).
37  * The format standard is maintained by the National
38  * Geophysical Data Center of NOAA.
39  * 2. The standard MGD77 format includes a 1920 byte header
40  * followed by 120 byte data records. The header consists
41  * of 24 80 byte records. The first character of the first
42  * header record is either 1 (pre-Y2K fix) or 4 (post-Y2K fix).
43  * MB-System treats the header as 16 120 byte records and
44  * provides no means of modifying the header.
45  * 3. The data records are each 120 bytes long. The first
46  * character of each data record is either
47  * 3 (pre-Y2K fix) or 5 (post-Y2K fix).
48  * 4. The MB-System implementation includes the support of
49  * an arbitrary number of comment records at the beginning
50  * of each file. The comment records are 120 bytes each
51  * and begin with the character '#'.
52  *
53  */

149  double mag_altitude;

164  double free_air; /* FREE-AIR ANOMALY
