35 /*
36  * Notes on the mbsys_wassp data structure and associated format:
37  * 1. This MBIO format supports the generic interface message output
38  * by the ENL WASSP multibeam sonars.
39  * 2. Reference: WASSP Generic ICD 2.2, 15 October 2013.
40  * 2. WASSP Generic ICD 2.4, 17 February 2015.
41  * 3. The WASSP data stream consists of several different data
42  * records that can vary among models and installations.
43  * 4. The WASSP multibeam models as of January 2014 incude:
44  * WMB-3230, 160 kHz, 112 beams, 120 degree swath
45  * WMB-5230, 80 kHz, 112 beams, 120 degree swath
46  * WMB-3250, 160 kHz, 224 beams, 120 degree swath
47  * 5. The alongtrack beamwidths are 4 degrees, the acrosstrack
48  * beamwidths are estimated to be 2 degrees since the transducer
49  * arrays are about two times wider than long.
50  * 6. Each data record begins with a 16 byte sequence:
51  * unsigned int sync; \\ 0x77F9345A
52  * unsigned int size; \\ Size in bytes of this record from start
53  * \\ of sync pattern to end of checksum
54  * char header[8]; \\ Data record name
55  * 7. All data are in little-endian form.
56  * 8. The data record names include:
57  * GENBATHY - Uncorrected Bathymetry
58  * GEN_SENS - External Sensor Data
59  * RAWSONAR - Raw water column data (roll stabilized sonar)
60  * CORBATHY - Corrected Bathymetry
61  * NVUPDATE - Nav Data Message
62  * WCD_NAVI - Water Column Information
63  * SYS_CFG1 - Unknown record at start of file
64  * (3224 bytes including sync and checksum)
65  * 9. Some but not all sample files logged by the ENL datalogger have an 804 byte UTF-8
66  * header that looks like:
67  * ********************************************************************************
68  * **************************************** PM Version: 2.4.1.288****************
69  * **************************************** GUI Version: 2.4.1.125****************
70  * **************************************** Mity Verion: 20060****************
71  * ********************************************************************************
72  * ********************************************************************************
73  * **************************************** PM Version: 2.4.1.288****************
74  * **************************************** GUI Version: 2.4.1.125****************
75  * **************************************** Mity Verion: 20060****************
76  * ********************************************************************************
77  * The reading code will search for valid sync values and so should ignore similar
78  * headers when encountered.
79  * A sample file logged by Jonathon Beaudoin of the University of New Hampshire
80  * Center for Coastal and Ocean Mapping does not contain this header, so it is a
81  * feature of the ENL datalogger and not part of the data stream output by the
82  * WASSP sonars.
83  * 10. The CORBATHY records do not have a full time stamp, and thus do not stand alone
84  * as survey data. Both a GENBATHY and a CORBATHY record are required to form a useful
85  * survey record.
86  * 11. A survey record may or may not include a RAWSONAR record and a WCD_NAVI record.
87  * 12. The UNH/CCOM data samples contain adjacent GENBATHY and CORBATHY records for each ping.
88  * The ENL-logged data have separate groups of GENBATHY records and CORBATHY records,
89  * with each CORBATHY record occurring 10 to 30 records after the corresponding
90  * GENBATHY record. Consequently, parsing these data would require buffering the GENBATHY
91  * records to match them with the right CORBATHY record. The current I/O module is
92  * not implemented with GENBATHY buffering, and will not work with the ENL data sample.
93  * 13. The contents of the GEN_SENS records are not specified and are presently unknown.
94  * 14. The NVUPDATE record does not include a time stamp and thus does not usefully serve
95  * as asynchronous navigation and attitude.
96  * 15. The maximum number of beams is known, and can be statically dimensioned for the
97  * GENBATHY and CORBATHY records. However, the maximum numbers of samples in the RAWSONAR
98  * and WCD_NAVI records are not defined, and so these structures must allow for
99  * dynamic allocation of memory.
100  *
101  */

136 struct mbsys_wassp_genbathy_struct {
137  /* GENBATHY Record */
138  /* Supported Products: - WMB-3250 */
139  /* Uncorrected Bathymetry
140  * All the bottom detection points will be supplied as range and angle values.
141  * The length of the output message is variable, dependent on the number of beam data.
142  * In addition to the Flags the sample number will be set to zero when detection is invalid. */
143  /* unsigned int sync; */ /* 0x77F9345A */
144  /* unsigned int size; */ /* Size in bytes of this record from start of sync pattern to end of checksum */
145  /* char header[8]; */ /* "GENBATHY" */
146  unsigned int version; /* 3 */
147  double msec; /* A millisecond time stamp of rising edge
148  * of Transmit pulse in UTC time (UTC time
149  * is calculated from the timestamp of the
150  * ZDA sentence and or a PPS signal when
151  * available) No local time zone correction
152  * is applied.
153  * On systems not supporting UTC time, this
154  * will be system referenced time. */
155  mb_u_char day; /* UTC time from NMEA ZDA time message, if available */
156  mb_u_char month; /* UTC time from NMEA ZDA time message, if available */
157  unsigned short year; /* UTC time from NMEA ZDA time message, if available */
158  unsigned int ping_number; /* Sequential number. */
159  unsigned int sonar_model; /* unused in ICD 2.4 */
160  unsigned long transducer_serial_number;
161  unsigned int number_beams;
162  unsigned int modeflags; /* Bit field:
163  * Bit 0: Roll information is valid (not implemented)
164  * Bit 1: Backscatter information is valid in beam data
165  * Bits 2-31: Reserved for future use */
166  float sampling_frequency; /* Sonar sampling frequency (Hz) */
167  float acoustic_frequency; /* Sonar nominal acoustic frequency (Hz) */
168  float tx_power; /* Voltage (volts) rms applied to
169  * transmitter in dB. If "Sample Type" field is set
170  * to "un-calibrated" this will be the nominal
171  * power level */
172  float pulse_width; /* Pulse length in milliseconds */
173  float absorption_loss; /* Configurable value applied by WASSP. */
174  float spreading_loss; /* 0, 30 or 40 as selected by WASSP GUI,
175  * units dB (as function of target range
176  * in metres) */
177  unsigned int sample_type; /* Set to 0 if un-calibrated. Set to 1 if calibrated.
178  * All dB values will be relative rather than absolute
179  * if ‘Sample Type’ is ‘un-calibrated’ */
180  float sound_velocity; /* Sound velocity at the sonar head in m/s
181  * (that was used in beam forming) */
182  float detection_point[MBSYS_WASSP_MAX_BEAMS];
183  /* Non-corrected fractional sample number
184  * with the reference to the receiver’s
185  * acoustic centre with the zero sample at
186  * the transmit time. */
187  float rx_angle[MBSYS_WASSP_MAX_BEAMS];
188  /* Beam steering angle with reference to
189  * receiver’s acoustic centre in the sonar
190  * reference frame, at the detection point;
191  * in degrees. */
192  unsigned int flags[MBSYS_WASSP_MAX_BEAMS];
193  /* Bit fields:
194  Bit 0: Detection Success
195  Bit 1: Detection Valid
196  Bit 2: WMT Detection
197  Bit 3: SAC Detection
198  Bits 4-31: Reserved for future use */
199  float backscatter[MBSYS_WASSP_MAX_BEAMS];
200  /* Target strength (dB) at seafloor
201  * detection point on this beam. */
202  unsigned int checksum; /* 0x8806CBA5 (not really a checksum, actually a sync value) */
203 };

230 struct mbsys_wassp_rawsonar_struct {
231  /* RAWSONAR Record */
232  /* Supported Products: - WMB-3250 */
233  /* Raw water column data (roll stabilized sonar)
234  * This packet is only roll stabilized if the WASSP system has valid roll information available.
235  * The data contained in this packet is to be dB signal levels received by time and angle. Some
236  * filtering of the data may be applied to remove side lobes and noise. This data is likely to be
237  * a subset of the full sampling resolution of the system – less beams and less samples. The first
238  * sample of raw data is the first sampling period starting from the rising edge of the transmit
239  * pulse and ending at the end of the sampling period determined by the Sample Rate.
240  * The Spreading Loss value applied to this data can be found in the preceding GENBATHY packet.
241  */

246  * ZDA sentence and or a PPS signal when
247  * available) No local time zone correction
248  * is applied.
249  * On systems not supporting UTC time, this
250  * will be system referenced time. */
251  unsigned int ping_number; /* Sequential number. */
252  float sample_rate; /* Frequency (Hz) of raw data in this packet */
253  unsigned int n; /* Number of beams of raw data in this packet */
254  unsigned int m; /* Number of samples (per beam) of raw data in this packet */
255  float tx_power; /* Voltage (volts) rms applied to
256  * transmitter in dB.*/
257  float pulse_width; /* Pulse length in milliseconds */
258  unsigned int sample_type; /* Set to 0 if un-calibrated.
259  * Set to 1 if calibrated. */
260  unsigned short spare[MBSYS_WASSP_MAX_BEAMS];
261  /* Unused */
262  unsigned short beam_index[MBSYS_WASSP_MAX_BEAMS];
263  /* Equivalent beam Index into uncorrected bathy
264  * (GENBATHY) record of each beam. */
265  unsigned int detection_point[MBSYS_WASSP_MAX_BEAMS];
266  /* Index of sample which most closely matches
267  * seafloor detection. 0 = not valid. */
268  float beam_angle[MBSYS_WASSP_MAX_BEAMS];
269  /* Beam angle for this beam in degrees (negative
270  * port side of nadir) */
271  size_t rawdata_alloc; /* Number of shorts allocated for rawdata array */
272  short *rawdata; /* If Sample Type = 0 then Signal Levels at
273  * sample/beam in dB*100 (divide by 100 to get
274  * actual signal level dB). The order is
275  * sample 1 x [0, 1, 2, 3, ... N]
276  * sample 2 x [0, 1, 2, 3, ... N]
277  * .....
278  * sample M x [0, 1, 2, 3, ... N]
279  * If Sample Type = 1 then calibrated db*100. */
280  unsigned int checksum; /* 0x8806CBA5 (not really a checksum, actually a sync value) */
281 };

283 struct mbsys_wassp_corbathy_struct {
284  /* CORBATHY Record */
285  /* Supported Products:
286  * WMB-3250
287  * WMB-3230
288  * WMB-5230 */
289  /* Corrected Bathymetry
290  * Use this data for use as fully corrected bathymetry data. The ships sensors
291  * integrated into the WASSP correct this information for leaver arm, pitch, roll, yaw,
292  * heave, tide etc. Each total message contains the detections data for a single ping.
293  * NOTES:
294  * 1) Sign of Latitude (N = +ve)
295  * 2) Sign of Longitude (E = +ve)
296  * 3) All points are sent for every beam even if they contain no detection data.
297  * So you can check if it is valid by checking the Y value, if this is 0 then the
298  * detection is not valid and should not be used.
299  * 4) The X,Y,Z positions are relative to the position in Table 5 Corrected Bathy.
300  * This is based on the fully corrected output using lever arm,
301  * sensor data and sound speed information available. This means if the X, Y, Z
302  * offsets in the WASSP application are correct, there is no need to account for
303  * the distance between GPS antenna and transducer or any pitch/roll/heave inclination.
304  * 5) Sign of Longitude is normal (East = positive)
305  * 6) Depths are tide corrected unless tides are disabled in the WASSP system.
306  */
307  /* unsigned int sync; */ /* 0x77F9345A */
308  /* unsigned int size; */ /* Size in bytes of this record from start of sync pattern to end of checksum */
309  /* char header[8]; */ /* "CORBATHY" */
310  unsigned int version; /* 3 for ICD 2.2, 4 for ICD 2.4 */
311  double msec; /* Version 4: unused
312  * Version 3:
313  * A millisecond time stamp of rising edge
314  * of Transmit pulse in UTC time (UTC time
315  * is calculated from the timestamp of the
316  * ZDA sentence and or a PPS signal when
317  * available) No local time zone correction
318  * is applied.
319  * On systems not supporting UTC time, this
320  * will be system referenced time. */
321  unsigned int num_beams; /* Fixed by software. Invalid points have depth set to 0.0. */
322  unsigned int ping_number; /* Ping sequence number */
323  double latitude; /* Latitude at transducer in degrees*/
324  double longitude; /* Longitude at transducer in degrees */
325  float bearing; /* Bearing/Heading of vessel on transmit in degrees */
326  float roll; /* Roll of vessel on transmit in degrees */
327  float pitch; /* Pitch of vessel on transmit in degrees */
328  float heave; /* Heave of vessel on transmit at transducer in meters */
329  unsigned int sample_type; /* Set to 0 if un-calibrated. Set to 1 if calibrated. */
330  float tide; /* Tide adjustment applied in meters */
331  unsigned int spare[5];
332  unsigned int beam_index[MBSYS_WASSP_MAX_BEAMS];
333  /* Beam index number */
334  float x[MBSYS_WASSP_MAX_BEAMS];
335  /* Distance to detection point in metres laterally along
336  * west/east axis. East of position is positive. */
337  float y[MBSYS_WASSP_MAX_BEAMS];
338  /* Distance to detection point in metres laterally along
339  * north/south axis. South of position is positive. */
340  float z[MBSYS_WASSP_MAX_BEAMS];
341  /* Depth in meters for the detection point. –ve = down.
342  * 0 = not valid */
343  float beam_angle[MBSYS_WASSP_MAX_BEAMS];
344  /* Angle of the beam this detection appears on in radians
345  * positive for starboard side of vessel. */
346  float backscatter[MBSYS_WASSP_MAX_BEAMS];
347  /* Post-processed target strength (dB) at seafloor on this beam. */
348  mb_u_char quality[MBSYS_WASSP_MAX_BEAMS];
349  /* Detection information - (0=none, 1=WMT, 2=SAC) */
350  mb_u_char fish[MBSYS_WASSP_MAX_BEAMS];
351  /* Fish intensity value for all fish targets vertically above
352  * detection point. */
353  mb_u_char roughness[MBSYS_WASSP_MAX_BEAMS]; /* Unused */
354  mb_u_char empty[MBSYS_WASSP_MAX_BEAMS]; /* Unused */
355  unsigned int pad[MBSYS_WASSP_MAX_BEAMS]; /* Unused */
356  unsigned int checksum; /* 0x8806CBA5 (not really a checksum, actually a sync value) */
357 };
